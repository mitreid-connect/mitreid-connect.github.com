020package org.mitre.openid.connect.service.impl;
021
022import java.util.Collection;
023
024import org.mitre.openid.connect.model.BlacklistedSite;
025import org.mitre.openid.connect.repository.BlacklistedSiteRepository;
026import org.mitre.openid.connect.service.BlacklistedSiteService;
027import org.springframework.beans.factory.annotation.Autowired;
028import org.springframework.stereotype.Service;
029import org.springframework.transaction.annotation.Transactional;
030
031import com.google.common.base.Strings;
032
033/**
034 * @author jricher
035 *
036 */
037@Service
038@Transactional
039public class DefaultBlacklistedSiteService implements BlacklistedSiteService {
040
041        @Autowired
042        private BlacklistedSiteRepository repository;
043
044        /* (non-Javadoc)
045         * @see org.mitre.openid.connect.service.BlacklistedSiteService#getAll()
046         */
047        @Override
048        public Collection<BlacklistedSite> getAll() {
049                return repository.getAll();
050        }
051
052        /* (non-Javadoc)
053         * @see org.mitre.openid.connect.service.BlacklistedSiteService#getById(java.lang.Long)
054         */
055        @Override
056        public BlacklistedSite getById(Long id) {
057                return repository.getById(id);
058        }
059
060        /* (non-Javadoc)
061         * @see org.mitre.openid.connect.service.BlacklistedSiteService#remove(org.mitre.openid.connect.model.BlacklistedSite)
062         */
063        @Override
064        public void remove(BlacklistedSite blacklistedSite) {
065                repository.remove(blacklistedSite);
066        }
067
068        /* (non-Javadoc)
069         * @see org.mitre.openid.connect.service.BlacklistedSiteService#saveNew(org.mitre.openid.connect.model.BlacklistedSite)
070         */
071        @Override
072        public BlacklistedSite saveNew(BlacklistedSite blacklistedSite) {
073                return repository.save(blacklistedSite);
074        }
075
076        /* (non-Javadoc)
077         * @see org.mitre.openid.connect.service.BlacklistedSiteService#update(org.mitre.openid.connect.model.BlacklistedSite, org.mitre.openid.connect.model.BlacklistedSite)
078         */
079        @Override
080        public BlacklistedSite update(BlacklistedSite oldBlacklistedSite, BlacklistedSite blacklistedSite) {
081                return repository.update(oldBlacklistedSite, blacklistedSite);
082        }
083
084        /* (non-Javadoc)
085         * @see org.mitre.openid.connect.service.BlacklistedSiteService#isBlacklisted(java.lang.String)
086         */
087        @Override
088        public boolean isBlacklisted(String uri) {
089
090                if (Strings.isNullOrEmpty(uri)) {
091                        return false; // can't be blacklisted if you don't exist
092                }
093
094                Collection<BlacklistedSite> sites = getAll();
095
096                // TODO: rewrite this to do regex matching and use the Guava predicates collection
097
098                for (BlacklistedSite blacklistedSite : sites) {
099                        if (Strings.nullToEmpty(blacklistedSite.getUri()).equals(uri)) {
100                                return true;
101                        }
102                }
103
104                return false;
105        }
106
107}
