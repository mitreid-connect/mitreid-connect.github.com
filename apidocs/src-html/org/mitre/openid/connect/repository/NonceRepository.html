001package org.mitre.openid.connect.repository;
002
003import java.util.Collection;
004import org.mitre.openid.connect.model.Nonce;
005
006/**
007 * 
008 * Nonce repository interface.
009 * 
010 * @author Amanda Anganes
011 *
012 */
013public interface NonceRepository {
014
015        /**
016         * Return the nonce with the given ID
017         * 
018         * @param id the ID of the nonce to find
019         * @return the nonce, if found
020         */
021        public Nonce getById(Long id);
022        
023        /**
024         * Remove the given Nonce from the database
025         * 
026         * @param nonce the Nonce to remove
027         */
028        public void remove(Nonce nonce);
029        
030        /**
031         * Save a new Nonce in the database
032         * 
033         * @param nonce the Nonce to save
034         * @return the saved Nonce
035         */
036        public Nonce save(Nonce nonce);
037        
038        /**
039         * Return all nonces stored in the database
040         * 
041         * @return the set of nonces
042         */
043        public Collection<Nonce> getAll();
044        
045        /**
046         * Return all expired nonces stored in the database
047         * 
048         * @return the set of expired nonces
049         */
050        public Collection<Nonce> getExpired();
051        
052        /**
053         * Return the set of nonces registered to the given client ID
054         * 
055         * @param clientId the client ID
056         * @return the set of nonces registered to the client
057         */
058        public Collection<Nonce> getByClientId(String clientId);
059        
060}
