004package org.mitre.openid.connect.client.keypublisher;
005
006import java.util.Locale;
007
008import org.springframework.core.Ordered;
009import org.springframework.web.servlet.View;
010import org.springframework.web.servlet.ViewResolver;
011
012/**
013 * 
014 * Simple view resolver to map JWK view names to appropriate beans
015 * 
016 * @author jricher
017 *
018 */
019public class JwkViewResolver implements ViewResolver, Ordered {
020
021        private String jwkViewName = "jwkKeyList";
022        private View jwk;
023        
024        private String x509ViewName = "x509certs";
025        private View x509;
026        
027        private int order = HIGHEST_PRECEDENCE; // highest precedence, most specific -- avoids hitting the catch-all view resolvers
028        
029        /**
030         * Map "jwkKeyList" to the jwk property and "x509certs" to the x509 property on this bean.
031         * Everything else returns null
032         */
033        @Override
034        public View resolveViewName(String viewName, Locale locale) throws Exception {
035                if (viewName != null) {
036                        if (viewName.equals(getJwkViewName())) {
037                                return getJwk();
038                        } else if (viewName.equals(getX509ViewName())) {
039                                return getX509();
040                        } else {
041                                return null;
042                        }
043                } else {
044                        return null;
045                }
046        }
047
048        /**
049     * @return the x509
050     */
051    public View getX509() {
052        return x509;
053    }
054
055        /**
056     * @param x509 the x509 to set
057     */
058    public void setX509(View x509) {
059        this.x509 = x509;
060    }
061
062        /**
063     * @return the jwk
064     */
065    public View getJwk() {
066        return jwk;
067    }
068
069        /**
070     * @param jwk the jwk to set
071     */
072    public void setJwk(View jwk) {
073        this.jwk = jwk;
074    }
075
076        /**
077     * @return the order
078     */
079    @Override
080    public int getOrder() {
081            return order;
082    }
083
084        /**
085     * @param order the order to set
086     */
087    public void setOrder(int order) {
088            this.order = order;
089    }
090
091        /**
092     * @return the jwkViewName
093     */
094    public String getJwkViewName() {
095            return jwkViewName;
096    }
097
098        /**
099     * @param jwkViewName the jwkViewName to set
100     */
101    public void setJwkViewName(String jwkViewName) {
102            this.jwkViewName = jwkViewName;
103    }
104
105        /**
106     * @return the x509ViewName
107     */
108    public String getX509ViewName() {
109            return x509ViewName;
110    }
111
112        /**
113     * @param x509ViewName the x509ViewName to set
114     */
115    public void setX509ViewName(String x509ViewName) {
116            this.x509ViewName = x509ViewName;
117    }
118
119}
