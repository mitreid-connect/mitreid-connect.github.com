004package org.mitre.openid.connect.client.model;
005
006/**
007 * 
008 * Data container to facilitate returns from the IssuerService API.
009 * 
010 * @author jricher
011 *
012 */
013public class IssuerServiceResponse {
014
015        private String issuer;
016        private String loginHint;
017        private String targetLinkUri;
018        private String redirectUrl;
019        
020        /**
021         * @param issuer
022         * @param loginHint
023         * @param targetLinkUri
024         */
025    public IssuerServiceResponse(String issuer, String loginHint, String targetLinkUri) {
026            this.issuer = issuer;
027            this.loginHint = loginHint;
028            this.targetLinkUri = targetLinkUri;
029    }
030    
031        /**
032         * @param redirectUrl
033         */
034    public IssuerServiceResponse(String redirectUrl) {
035            this.redirectUrl = redirectUrl;
036    }
037        /**
038         * @return the issuer
039         */
040        public String getIssuer() {
041                return issuer;
042        }
043        /**
044         * @param issuer the issuer to set
045         */
046        public void setIssuer(String issuer) {
047                this.issuer = issuer;
048        }
049        /**
050         * @return the loginHint
051         */
052        public String getLoginHint() {
053                return loginHint;
054        }
055        /**
056         * @param loginHint the loginHint to set
057         */
058        public void setLoginHint(String loginHint) {
059                this.loginHint = loginHint;
060        }
061        /**
062         * @return the targetLinkUri
063         */
064        public String getTargetLinkUri() {
065                return targetLinkUri;
066        }
067        /**
068         * @param targetLinkUri the targetLinkUri to set
069         */
070        public void setTargetLinkUri(String targetLinkUri) {
071                this.targetLinkUri = targetLinkUri;
072        }
073        /**
074         * @return the redirectUrl
075         */
076        public String getRedirectUrl() {
077                return redirectUrl;
078        }
079        /**
080         * @param redirectUrl the redirectUrl to set
081         */
082        public void setRedirectUrl(String redirectUrl) {
083                this.redirectUrl = redirectUrl;
084        }
085        
086        /**
087         * If the redirect url has been set, then we should send a redirect using it instead of processing things.
088         */
089        public boolean shouldRedirect() {
090                return this.redirectUrl != null;
091        }
092        
093}
