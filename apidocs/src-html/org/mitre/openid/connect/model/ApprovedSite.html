017package org.mitre.openid.connect.model;
018
019import java.util.Date;
020import java.util.Set;
021
022import javax.persistence.Basic;
023import javax.persistence.CollectionTable;
024import javax.persistence.Column;
025import javax.persistence.ElementCollection;
026import javax.persistence.Entity;
027import javax.persistence.FetchType;
028import javax.persistence.GeneratedValue;
029import javax.persistence.GenerationType;
030import javax.persistence.Id;
031import javax.persistence.JoinColumn;
032import javax.persistence.ManyToOne;
033import javax.persistence.NamedQueries;
034import javax.persistence.NamedQuery;
035import javax.persistence.Table;
036import javax.persistence.Temporal;
037import javax.persistence.Transient;
038
039@Entity
040@Table(name="approved_site")
041@NamedQueries({
042        @NamedQuery(name = "ApprovedSite.getAll", query = "select a from ApprovedSite a"),
043        @NamedQuery(name = "ApprovedSite.getByUserId", query = "select a from ApprovedSite a where a.userId = :userId"),
044        @NamedQuery(name = "ApprovedSite.getByClientId", query = "select a from ApprovedSite a where a.clientId = :clientId"),
045        @NamedQuery(name = "ApprovedSite.getByClientIdAndUserId", query = "select a from ApprovedSite a where a.clientId = :clientId and a.userId = :userId")
046})
047public class ApprovedSite {
048
049        // unique id
050        private Long id;
051
052        // which user made the approval
053        private String userId;
054
055        // which OAuth2 client is this tied to
056        private String clientId;
057
058        // when was this first approved?
059        private Date creationDate;
060
061        // when was this last accessed?
062        private Date accessDate;
063
064        // if this is a time-limited access, when does it run out?
065        private Date timeoutDate;
066
067        // what scopes have been allowed
068        // this should include all information for what data to access
069        private Set<String> allowedScopes;
070
071        // If this AP is a WS, link to the WS
072        private WhitelistedSite whitelistedSite;
073
074        // TODO: should we store the OAuth2 tokens and IdTokens here?
075
076        /**
077         * Empty constructor
078         */
079        public ApprovedSite() {
080
081        }
082
083        /**
084         * @return the id
085         */
086        @Id
087        @GeneratedValue(strategy = GenerationType.IDENTITY)
088        @Column(name = "id")
089        public Long getId() {
090                return id;
091        }
092
093        /**
094         * @param id the id to set
095         */
096        public void setId(Long id) {
097                this.id = id;
098        }
099
100        /**
101         * @return the userInfo
102         */
103        @Basic
104        @Column(name="user_id")
105        public String getUserId() {
106                return userId;
107        }
108
109        /**
110         * @param userInfo the userInfo to set
111         */
112        public void setUserId(String userId) {
113                this.userId = userId;
114        }
115
116        /**
117         * @return the clientId
118         */
119        @Basic
120        @Column(name="client_id")
121        public String getClientId() {
122                return clientId;
123        }
124
125        /**
126         * @param clientId the clientId to set
127         */
128        public void setClientId(String clientId) {
129                this.clientId = clientId;
130        }
131
132        /**
133         * @return the creationDate
134         */
135        @Basic
136        @Temporal(javax.persistence.TemporalType.TIMESTAMP)
137        @Column(name="creation_date")
138        public Date getCreationDate() {
139                return creationDate;
140        }
141
142        /**
143         * @param creationDate the creationDate to set
144         */
145        public void setCreationDate(Date creationDate) {
146                this.creationDate = creationDate;
147        }
148
149        /**
150         * @return the accessDate
151         */
152        @Basic
153        @Temporal(javax.persistence.TemporalType.TIMESTAMP)
154        @Column(name="access_date")
155        public Date getAccessDate() {
156                return accessDate;
157        }
158
159        /**
160         * @param accessDate the accessDate to set
161         */
162        public void setAccessDate(Date accessDate) {
163                this.accessDate = accessDate;
164        }
165
166        /**
167         * @return the allowedScopes
168         */
169        @ElementCollection(fetch = FetchType.EAGER)
170        @CollectionTable(
171                        name="approved_site_scope",
172                        joinColumns=@JoinColumn(name="owner_id")
173                        )
174        @Column(name="scope")
175        public Set<String> getAllowedScopes() {
176                return allowedScopes;
177        }
178
179        /**
180         * @param allowedScopes the allowedScopes to set
181         */
182        public void setAllowedScopes(Set<String> allowedScopes) {
183                this.allowedScopes = allowedScopes;
184        }
185
186        /**
187         * @return the timeoutDate
188         */
189        @Basic
190        @Temporal(javax.persistence.TemporalType.TIMESTAMP)
191        @Column(name="timeout_date")
192        public Date getTimeoutDate() {
193                return timeoutDate;
194        }
195
196        /**
197         * @param timeoutDate the timeoutDate to set
198         */
199        public void setTimeoutDate(Date timeoutDate) {
200                this.timeoutDate = timeoutDate;
201        }
202
203        /**
204         * Does this AP entry correspond to a WS?
205         * @return
206         */
207        @Transient
208        public Boolean getIsWhitelisted() {
209                return (whitelistedSite != null);
210        }
211
212
213        @ManyToOne
214        @JoinColumn(name="whitelisted_site_id")
215        public WhitelistedSite getWhitelistedSite() {
216                return whitelistedSite;
217        }
218
219        public void setWhitelistedSite(WhitelistedSite whitelistedSite) {
220                this.whitelistedSite = whitelistedSite;
221        }
222
223        /**
224         * Has this approval expired?
225         * @return
226         */
227        @Transient
228        public boolean isExpired() {
229                if (getTimeoutDate() != null) {
230                        Date now = new Date();
231                        if (now.after(getTimeoutDate())) {
232                                return true;
233                        } else {
234                                return false;
235                        }
236                } else {
237                        return false;
238                }
239        }
240
241}
