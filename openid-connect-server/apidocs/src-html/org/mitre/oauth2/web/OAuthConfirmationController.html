061@Controller
062@SessionAttributes("authorizationRequest")
063public class OAuthConfirmationController {
064
065        @Autowired
066        private ClientDetailsEntityService clientService;
067
068        @Autowired
069        private SystemScopeService scopeService;
070
071        @Autowired
072        private ScopeClaimTranslationService scopeClaimTranslationService;
073
074        @Autowired
075        private UserInfoService userInfoService;
076
077        @Autowired
078        private StatsService statsService;
079
080        private static Logger logger = LoggerFactory.getLogger(OAuthConfirmationController.class);
081
082        public OAuthConfirmationController() {
083
084        }
085
086        public OAuthConfirmationController(ClientDetailsEntityService clientService) {
087                this.clientService = clientService;
088        }
089
090        @PreAuthorize("hasRole('ROLE_USER')")
091        @RequestMapping("/oauth/confirm_access")
092        public String confimAccess(Map<String, Object> model, @ModelAttribute("authorizationRequest") AuthorizationRequest authRequest,
093                        Principal p) {
094
095                // Check the "prompt" parameter to see if we need to do special processing
096
097                String prompt = (String)authRequest.getExtensions().get("prompt");
098                List<String> prompts = Splitter.on(" ").splitToList(Strings.nullToEmpty(prompt));
099                if (prompts.contains("none")) {
100                        // we're not supposed to prompt, so "return an error"
101                        logger.info("Client requested no prompt, returning 403 from confirmation endpoint");
102                        model.put("code", HttpStatus.FORBIDDEN);
103                        return HttpCodeView.VIEWNAME;
104                }
105                
106                if (prompts.contains("consent")) {
107                        model.put("consent", true);
108                }
109
110                //AuthorizationRequest clientAuth = (AuthorizationRequest) model.remove("authorizationRequest");
111
112                ClientDetailsEntity client = null;
113
114                try {
115                        client = clientService.loadClientByClientId(authRequest.getClientId());
116                } catch (OAuth2Exception e) {
117                        logger.error("confirmAccess: OAuth2Exception was thrown when attempting to load client", e);
118                        model.put("code", HttpStatus.BAD_REQUEST);
119                        return HttpCodeView.VIEWNAME;
120                } catch (IllegalArgumentException e) {
121                        logger.error("confirmAccess: IllegalArgumentException was thrown when attempting to load client", e);
122                        model.put("code", HttpStatus.BAD_REQUEST);
123                        return HttpCodeView.VIEWNAME;
124                }
125
126                if (client == null) {
127                        logger.error("confirmAccess: could not find client " + authRequest.getClientId());
128                        model.put("code", HttpStatus.NOT_FOUND);
129                        return HttpCodeView.VIEWNAME;
130                }
131
132                model.put("auth_request", authRequest);
133                model.put("client", client);
134
135                String redirect_uri = authRequest.getRedirectUri();
136
137                model.put("redirect_uri", redirect_uri);
138
139
140                // pre-process the scopes
141                Set<SystemScope> scopes = scopeService.fromStrings(authRequest.getScope());
142
143                Set<SystemScope> sortedScopes = new LinkedHashSet<SystemScope>(scopes.size());
144                Set<SystemScope> systemScopes = scopeService.getAll();
145
146                // sort scopes for display based on the inherent order of system scopes
147                for (SystemScope s : systemScopes) {
148                        if (scopes.contains(s)) {
149                                sortedScopes.add(s);
150                        }
151                }
152
153                // add in any scopes that aren't system scopes to the end of the list
154                sortedScopes.addAll(Sets.difference(scopes, systemScopes));
155
156                model.put("scopes", sortedScopes);
157
158                // get the userinfo claims for each scope
159                UserInfo user = userInfoService.getByUsername(p.getName());
160                Map<String, Map<String, String>> claimsForScopes = new HashMap<String, Map<String, String>>();
161                if (user != null) {
162                        JsonObject userJson = user.toJson();
163        
164                        for (SystemScope systemScope : sortedScopes) {
165                                Map<String, String> claimValues = new HashMap<String, String>();
166        
167                                Set<String> claims = scopeClaimTranslationService.getClaimsForScope(systemScope.getValue());
168                                for (String claim : claims) {
169                                        if (userJson.has(claim) && userJson.get(claim).isJsonPrimitive()) {
170                                                // TODO: this skips the address claim
171                                                claimValues.put(claim, userJson.get(claim).getAsString());
172                                        }
173                                }
174        
175                                claimsForScopes.put(systemScope.getValue(), claimValues);
176                        }
177                }
178
179                model.put("claims", claimsForScopes);
180
181                // client stats
182                Integer count = statsService.getCountForClientId(client.getId());
183                model.put("count", count);
184
185
186                // contacts
187                if (client.getContacts() != null) {
188                        String contacts = Joiner.on(", ").join(client.getContacts());
189                        model.put("contacts", contacts);
190                }
191
192                // if the client is over a week old and has more than one registration, don't give such a big warning
193                // instead, tag as "Generally Recognized As Safe (gras)
194                Date lastWeek = new Date(System.currentTimeMillis() - (60 * 60 * 24 * 7 * 1000));
195                if (count > 1 && client.getCreatedAt() != null && client.getCreatedAt().before(lastWeek)) {
196                        model.put("gras", true);
197                } else {
198                        model.put("gras", false);
199                }
200
201                // inject a random value for CSRF purposes
202                model.put("csrf", authRequest.getExtensions().get("csrf"));
203
204                return "approve";
205        }
206
207        /**
208         * @return the clientService
209         */
210        public ClientDetailsEntityService getClientService() {
211                return clientService;
212        }
213
214        /**
215         * @param clientService the clientService to set
216         */
217        public void setClientService(ClientDetailsEntityService clientService) {
218                this.clientService = clientService;
219        }
