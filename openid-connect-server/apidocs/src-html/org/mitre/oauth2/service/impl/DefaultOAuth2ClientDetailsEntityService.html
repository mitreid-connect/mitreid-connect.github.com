016package org.mitre.oauth2.service.impl;
017
018import java.math.BigInteger;
019import java.security.SecureRandom;
020import java.util.Collection;
021import java.util.Date;
022import java.util.UUID;
023
024import org.apache.commons.codec.binary.Base64;
025import org.mitre.oauth2.model.ClientDetailsEntity;
026import org.mitre.oauth2.repository.OAuth2ClientRepository;
027import org.mitre.oauth2.repository.OAuth2TokenRepository;
028import org.mitre.oauth2.service.ClientDetailsEntityService;
029import org.mitre.openid.connect.model.WhitelistedSite;
030import org.mitre.openid.connect.service.ApprovedSiteService;
031import org.mitre.openid.connect.service.BlacklistedSiteService;
032import org.mitre.openid.connect.service.WhitelistedSiteService;
033import org.springframework.beans.factory.annotation.Autowired;
034import org.springframework.security.oauth2.common.exceptions.InvalidClientException;
035import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
036import org.springframework.stereotype.Service;
037
038import com.google.common.base.Strings;
039
040@Service
041public class DefaultOAuth2ClientDetailsEntityService implements ClientDetailsEntityService {
042        
043        @Autowired
044        private OAuth2ClientRepository clientRepository;
045        
046        @Autowired
047        private OAuth2TokenRepository tokenRepository;
048        
049        @Autowired
050        private ApprovedSiteService approvedSiteService;
051        
052        @Autowired
053        private WhitelistedSiteService whitelistedSiteService;
054        
055        @Autowired
056        private BlacklistedSiteService blacklistedSiteService;
057        
058        public DefaultOAuth2ClientDetailsEntityService() {
059                
060        }
061        
062        public DefaultOAuth2ClientDetailsEntityService(OAuth2ClientRepository clientRepository, 
063                        OAuth2TokenRepository tokenRepository) {
064                this.clientRepository = clientRepository;
065                this.tokenRepository = tokenRepository;
066        }
067        
068        @Override
069        public ClientDetailsEntity saveNewClient(ClientDetailsEntity client) {
070                if (client.getId() != null) { // if it's not null, it's already been saved, this is an error
071                        throw new IllegalArgumentException("Tried to save a new client with an existing ID: " + client.getId());
072                }
073
074                if (client.getRegisteredRedirectUri() != null) {
075                        for (String uri : client.getRegisteredRedirectUri()) {
076                                if (blacklistedSiteService.isBlacklisted(uri)) {
077                                        throw new IllegalArgumentException("Client URI is blacklisted: " + uri);
078                                }
079                    }
080                }
081                
082                // assign a random clientid if it's empty 
083                // NOTE: don't assign a random client secret without asking, since public clients have no secret
084        if (Strings.isNullOrEmpty(client.getClientId())) {
085            client = generateClientId(client);
086        }
087        
088        // if the client is flagged to allow for refresh tokens, make sure it's got the right granted scopes
089        if (client.isAllowRefresh()) {
090                client.getScope().add("offline_access");
091        } else {
092                client.getScope().remove("offline_access");
093        }
094        
095        // timestamp this to right now
096        client.setCreatedAt(new Date());
097        
098        return clientRepository.saveClient(client);
099        }
100        
101        /**
102         * Get the client by its internal ID
103         */
104        public ClientDetailsEntity getClientById(Long id) {
105                ClientDetailsEntity client = clientRepository.getById(id);
106                
107                return client;
108        }
109        
110        /**
111         * Get the client for the given ClientID
112         */
113        @Override
114        public ClientDetailsEntity loadClientByClientId(String clientId) throws OAuth2Exception, InvalidClientException, IllegalArgumentException {
115                if (!Strings.isNullOrEmpty(clientId)) {
116                        ClientDetailsEntity client = clientRepository.getClientByClientId(clientId);
117                        if (client == null) {
118                                throw new InvalidClientException("Client with id " + clientId + " was not found");
119                        }
120                        else {
121                                return client;
122                        }
123                }
124                
125                throw new IllegalArgumentException("Client id must not be empty!");
126        }
127        
128        /**
129         * Delete a client and all its associated tokens
130         */
131        @Override
132    public void deleteClient(ClientDetailsEntity client) throws InvalidClientException {
133                
134                if (clientRepository.getById(client.getId()) == null) {
135                        throw new InvalidClientException("Client with id " + client.getClientId() + " was not found");
136                }
137                
138                // clean out any tokens that this client had issued
139                tokenRepository.clearTokensForClient(client);
140                
141                // clean out any approved sites for this client
142                approvedSiteService.clearApprovedSitesForClient(client);
143                
144                // clear out any whitelisted sites for this client
145                WhitelistedSite whitelistedSite = whitelistedSiteService.getByClientId(client.getClientId());
146                if (whitelistedSite != null) {
147            whitelistedSiteService.remove(whitelistedSite);
148                }
149                
150                // take care of the client itself
151                clientRepository.deleteClient(client);
152                
153        }
154
155        /**
156         * Update the oldClient with information from the newClient. The 
157         * id from oldClient is retained.
158         */
159        @Override
160    public ClientDetailsEntity updateClient(ClientDetailsEntity oldClient, ClientDetailsEntity newClient) throws IllegalArgumentException {
161                if (oldClient != null && newClient != null) {
162                        
163                        for (String uri : newClient.getRegisteredRedirectUri()) {
164                                if (blacklistedSiteService.isBlacklisted(uri)) {
165                                        throw new IllegalArgumentException("Client URI is blacklisted: " + uri);
166                                }
167                }
168                        
169                // if the client is flagged to allow for refresh tokens, make sure it's got the right scope
170                if (newClient.isAllowRefresh()) {
171                        newClient.getScope().add("offline_access");
172                } else {
173                        newClient.getScope().remove("offline_access");
174                }
175
176                return clientRepository.updateClient(oldClient.getId(), newClient);
177                }
178                throw new IllegalArgumentException("Neither old client or new client can be null!");
179    }
180
181        /**
182         * Get all clients in the system
183         */
184        @Override
185    public Collection<ClientDetailsEntity> getAllClients() {
186                return clientRepository.getAllClients();
187    }
188
189        /**
190         * Generates a clientId for the given client and sets it to the client's clientId field. Returns the client that was passed in, now with id set.
191         */
192        @Override
193    public ClientDetailsEntity generateClientId(ClientDetailsEntity client) {
194                client.setClientId(UUID.randomUUID().toString());
195            return client;
196    }
197
198        /**
199         * Generates a new clientSecret for the given client and sets it to the client's clientSecret field. Returns the client that was passed in, now with secret set.
200         */
201        @Override
202    public ClientDetailsEntity generateClientSecret(ClientDetailsEntity client) {
203                client.setClientSecret(Base64.encodeBase64URLSafeString(new BigInteger(512, new SecureRandom()).toByteArray()).replace("=", ""));
204            return client;
205    }
